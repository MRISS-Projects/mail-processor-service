14  @Service
15  public class Validator {
16  
17      private static final Logger LOGGER = LoggerFactory.getLogger(Validator.class);
18  
19      @Value("${ALLOWED_ACCOUNT}")
20      private String allowedAccount;
21  
22      @Value("${ALLOWED_SENDER_LIST}")
23      private String allowedSenderList;
24  
25      @Value("${GAE_SERVICE}")
26      private String serviceName;
27  
28      @Value("${GOOGLE_CLOUD_PROJECT}")
29      private String projectId;
30  
31      private Pattern pattern;
32  
33      public Validator() {
34      }
35  
36      @PostConstruct
37      public void init() {
38          StringBuffer sb = new StringBuffer("/_ah/mail/");
39          this.pattern = Pattern.compile(sb.append("(").append(allowedAccount).append(")").append("@").append("(")
40                  .append(serviceName).append(")").append("(\\-dot\\-)").append("(").append(projectId).append(")")
41                  .append("(").append("\\.appspotmail\\.com").append(")").toString());
42      }
43  
44      public boolean validate(String requestURI, Set<String> set) {
45          Matcher m = pattern.matcher(requestURI);
46          return m.matches();
47      }
