33  @WebFilter("/_ah/mail/*")
34  public class RequestResponseLoggingFilter implements Filter {
35  
36      private static final Logger LOGGER = LoggerFactory.getLogger(RequestResponseLoggingFilter.class);
37  
38      @Override
39      public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
40              throws IOException, ServletException {
41          LOGGER.info("Entering filter...");
42          HttpServletRequest req = (HttpServletRequest) request;
43          HttpServletResponse res = (HttpServletResponse) response;
44          LOGGER.info("Logging Request  {} : {}", req.getMethod(), req.getRequestURI());
45          chain.doFilter(request, response);
46          LOGGER.info("Logging Response :{}", res.getContentType());
47          MimeMessage msg = getMessageFromRequest(req);
48          try {
49              LOGGER.info("Message subject: " + msg.getSubject());
50              LOGGER.info("Message subject: " + msg.getContent());
51              Multipart multipart = (Multipart) msg.getContent();
52              LOGGER.info("Multipart count: " + multipart.getCount());
53              for (int i = 0; i < multipart.getCount(); i++) {
54                  BodyPart bodyPart = multipart.getBodyPart(i);
55                  if (bodyPart != null) {
56                      try (InputStream is = bodyPart.getInputStream(); StringWriter sw = new StringWriter()) {
57                          IOUtils.copy(is, sw, StandardCharsets.UTF_8);
58                          StringBuffer buffer = sw.getBuffer();
59                          LOGGER.info("Content: " + buffer.toString());
60                      }
61                  }
62              }
63          } catch (MessagingException e) {
64              LOGGER.error(e.getMessage(), e);
65          } catch (IOException e) {
66              LOGGER.error(e.getMessage(), e);
67          }
68      }
69  
70      protected MimeMessage getMessageFromRequest(ServletRequest req) throws ServletException {
71          MimeMessage message = (MimeMessage) req.getAttribute("mimeMessage");
72          if (message == null) {
73              try {
74                  Properties props = new Properties();
75                  Session session = Session.getDefaultInstance(props, null);
76                  message = new MimeMessage(session, req.getInputStream());
77                  req.setAttribute("mimeMessage", message);
78              } catch (MessagingException e) {
79                  throw new ServletException("Error processing inbound message", e);
80              } catch (IOException e) {
81                  throw new ServletException("Error processing inbound message", e);
82              }
83          }
84          return message;
85      }
